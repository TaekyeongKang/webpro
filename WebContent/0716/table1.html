		  - tr(row) : 행, 줄
		  - td(data) : 열, 칸
		  - th(header) : td 대신 제목 표시
		  
		스타일
		  - table : border, border-collapse
		  - td : width, height, text-align
		  - tr : background
		  - th : width, height
